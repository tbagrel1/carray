219 void cvector_addi(cvector *p_cvector, value_t value, index_t index) {
220  if (index < 0) {
221  index = p_cvector->_size - index;
222  }
223  if (index > p_cvector->_size) {
224  CVECTOR_ERROR(0, "IndexOutOfBounds: cannot add an item after the end "
225  "of the array");
226  return;
227  }
228  if (index < 0) {
229  CVECTOR_ERROR(0, "IndexOutOfBounds: invalid (negative) index");
230  }
231  if (p_cvector->_size < p_cvector->_space) {
232  if (index == p_cvector->_size) {
233  cvector_add(p_cvector, value);
234  } else {
235  for (index_t i = p_cvector->_size - 1; i >= index; i--) {
236  p_cvector->_vector[i + 1] = p_cvector->_vector[i];
237  }
238  p_cvector->_size++;
239  p_cvector->_vector[index] = value;
240  }
241  } else {
242  cvector_addspace(p_cvector);
243  cvector_addi(p_cvector, value, index);

708  if (from < 0) {
709  from += p_cvector->_size;
710  }
711  if (to < 0) {
712  to += p_cvector->_size;
713  }
714  if (from < 0 || to < 0) {
715  CVECTOR_ERROR(0, "IndexOutOfBounds: invalid (negative) index");
716  return NULL;
717  }
718  if (from > p_cvector->_size - 1 || to > p_cvector->_size) {
719  CVECTOR_ERROR(0, "IndexOutOfBounds: invalid (out of bounds) index");
720  return NULL;
721  }
722  if (from > to) {
723  CVECTOR_ERROR(0, "IndexOutOfBounds: invalid index, from > to");
724  return NULL;
725  }
726  if (step <= 0) {
727  CVECTOR_ERROR(0, "InvalidStep: invalid (negative or null) step");
728  return NULL;
729  }
730  index_t size;
731  if (to == from) {
732  size = 0;
733  } else {
734  size = 1 + ((to - from - 1) / step); // Integer division there
735  }
736  cvector

738  cvector_new_space(ROUND_INDEX(CVECTOR_INIT_FACTOR * size + 1));
739  for (index_t i = 0; i < size; i++) {
740  p_result->_vector[i] = p_cvector->_vector[from + i * step];
741  }
742  p_result->_size = size;
743  return p_result;
